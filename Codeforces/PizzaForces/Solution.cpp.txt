#include <bits/stdc++.h>
#include <iostream>
#include <vector>
 using namespace std;
int main()
{
ios_base::sync_with_stdio(false);
cin.tie(NULL);
int t;
cin>>t;
while(t--)
 {
    long long int n,x,y;
     cin>>n;
     x=n/6;
     y=n%6;
     if(n<=6)
     {
         cout<<"15"<<endl;
     }
     else if(y==0)
     {
     cout<<x*15<<endl;
     }
     else if(y<=2)
     {
     cout<<(x*15)+5<<endl;
     }
     else if(y<=4)
     {
     cout<<(x*15)+10<<endl;
     }
     else{
         x++;
         cout<<x*15<<endl;
     }
 
 }
}
